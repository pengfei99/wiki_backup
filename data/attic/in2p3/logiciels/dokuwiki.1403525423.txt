<code php lib/plugins/certlogin/auth.php>
<?php
/**
 * DokuWiki Plugin certlogin (Auth Component)
 *
 * @license GPL 2 http://www.gnu.org/licenses/gpl-2.0.html
 * @author  Tanguy LE CARROUR <tanguy.lecarrour@bioaster.org>
 */

// must be run within Dokuwiki
if(!defined('DOKU_INC')) die();

class auth_plugin_certlogin extends DokuWiki_Auth_Plugin {

    /**
     * Constructor.
     */
    public function __construct() {
        parent::__construct(); // for compatibility

        $this->cando['external']    = true; // does the module do external auth checking?
        $this->cando['logout']      = true; // can the user logout again? (eg. not possible with HTTP auth)

        // FIXME intialize your auth system and set success to true, if successful
        $this->success = true;
    }

    /**
     * Log off the current user [ OPTIONAL ]
     */
    public function logOff() {
    }

    /**
     * Do all authentication [ OPTIONAL ]
     *
     * @param   string  $user    Username
     * @param   string  $pass    Cleartext Password
     * @param   bool    $sticky  Cookie should not expire
     * @return  bool             true on successful auth
     */
    public function trustExternal($user, $pass, $sticky = false) {

        global $USERINFO;
        global $conf;
        $sticky ? $sticky = true : $sticky = false; //sanity check

        // do the checking here

        $name = array_shift(explode(",", $_SERVER["SSL_CLIENT_S_DN_CN"]));
        $email = array_shift(explode(",", $_SERVER["SSL_CLIENT_S_DN_EMAILADDRESS"]));
        $pattern = '/@bioaster.org$/';
        if (preg_match($pattern, $email) == 0) {
            $USERINFO['name'] = "anonymous";
            $USERINFO['mail'] = "";
            $USERINFO['grps'] = array('ALL');
        }
        else {
            $USERINFO['name'] = $name;
            $USERINFO['mail'] = $email;
            $USERINFO['grps'] = array('admin', 'user');
        }

        $_SERVER['REMOTE_USER'] = $email;
        $_SESSION[DOKU_COOKIE]['auth']['user'] = $user;
        $_SESSION[DOKU_COOKIE]['auth']['pass'] = $pass;
        $_SESSION[DOKU_COOKIE]['auth']['info'] = $USERINFO;
        return true;
    }

    /**
     * Return user info
     *
     * Returns info about the given user needs to contain
     * at least these fields:
     *
     * name string  full name of the user
     * mail string  email addres of the user
     * grps array   list of groups the user is in
     *
     * @param   string $user the user name
     * @return  array containing user data or false
     */
    public function getUserData($user) {
        return $USERINFO;
    }

}

===== Connexion à l'AD =====

D'après le fichier de configuration du wiki de DEUZZI, le plugin ''authad'' a été utilisé avec les paramètres sont les suivants :

<code text local.protected.php>
<?php
  // general DokuWiki options
  $conf['useacl']         = 1;
  $conf['authtype']       = 'plain';
 
  // configure your Active Directory data here
  $conf['auth']['ad']['account_suffix']     = '@bioaster.local';
  $conf['auth']['ad']['base_dn']            = 'DC=bioaster,DC=local';
  $conf['auth']['ad']['domain_controllers'] = 'srvadmin.bioaster.local, srvfic.bioaster.local';
  $conf['plugin']['authad']['sso'] = 1;
  $conf['plugin']['authad']['admin_username'] = 'dokuwiki';
  $conf['plugin']['authad']['admin_password'] = 'XXXXXXXX';

  // optional parameters
  $conf['plugin']['authad']['recursive_groups']   = 1;
  $conf['plugin']['authad']['sso']                = 1;
  $conf['plugin']['authad']['real_primarygroup']  = 1;
  $conf['manager']   = '@DokuwikiAdmin';
  $conf['superuser'] = '@DokuwikiAdmin';
</code>
